堆

定义

堆是一个完全二叉树，但满足以下性质

  • 最大堆（Max Heap）：父节点的值大于等于子节点的值，根节点是最大值。
  • 最小堆（Min Heap）：父节点的值小于等于子节点的值，根节点是最小值。

    • 存储关系

    当堆存储在数组中, 假设当前节点的下标为i，则：

  • 左子节点的下标：2i + 1
  • 右子节点的下标：2i + 2
  • 父节点的下标：(i - 1) / 2

    • 基础操作

  • 插入操作（Insert）:将元素添加到堆的末尾，然后使用上浮操作（Bubble Up 或 Swim）
  • 删除操作（Delete）:将堆顶元素与堆的最后一个元素交换，然后使用下沉操作（Heapify Down 或 Sink）
  • 构建堆（Build Heap）:
    1.自底向上：从数组的最后一个非叶节点开始，逐个向前进行下沉操作。适合从无序数组构建堆，复杂度为O(n)。
    2.自顶向下：通常用于插入法，复杂度为O(n log n)

    • 上浮与下沉

    
public class Heap
{
    private List heap;

    public Heap()
    {
        heap = new List();
    }

    // 上浮
    public void Swim(List<int> heap, int index)
    {
        int parent = (index - 1) / 2;
        while (index > 0 && heap[index] > heap[parent])
        {
            int temp = heap[index];
            heap[index] = heap[parent];
            heap[parent] = temp;

            index = parent;
            parent = (index - 1) / 2;
        }
    }
    
    // 下沉
    public void Sink(List<int> heap, int index)
    {
        int n = heap.Count;
        while(true)
        {
            int left = 2 * index + 1;
            int right = 2 * index + 2;
            int largest = index;

            if (left < n && heap[left] > heap[largest]) 
            {
                largest = left;
            }

            if (right < n && heap[right] > heap[largest]) 
            {
                largest = right;
            }

            if (largest != index)
            {
                // 交换当前节点和较大的子节点
                int temp = heap[index];
                heap[index] = heap[largest];
                heap[largest] = temp;

                // 更新索引，继续向下调整
                index = largest;
            }
            else
            {
                break;
            }
        }
    }
}

    删除操作

    

    public  int Delete()
    {
        if (heap.Count == 0) return -1;

        // 交换堆顶元素和最后一个元素
        int top = heap[0];
        heap[0] = heap[heap.Count - 1];
        heap.RemoveAt(heap.Count - 1); // 删除最后一个元素

        Sink(heap, 0); // 调整堆

        return top; // 返回删除的堆顶元素
    }

    构建堆

    
    // 自底向上
    public void BuildHeap(List<int> heap)
    {
        int n = heap.Count;
        
        for (int i = n / 2 - 1; i >= 0; i--)
        {
            Sink(heap, i); 
        }
    }

    // 自顶向下
    public void BuildHeap(List<int> heap)
    {
        int n = heap.Count;

        for (int i = 0; i < n; i++)
        {
            Swim(heap, i); 
        }
    }
